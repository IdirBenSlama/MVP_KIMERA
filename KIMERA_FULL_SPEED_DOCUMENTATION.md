```sql
-- Geoids Table
geoids (
  geoid_id VARCHAR PRIMARY KEY,
  semantic_state_json JSON,
  symbolic_state JSON,
  semantic_vector FLOAT[],
  created_at TIMESTAMP,
  last_updated TIMESTAMP
)

-- SCARs Table  
scars (
  scar_id VARCHAR PRIMARY KEY,
  geoids JSON,
  reason TEXT,
  timestamp TIMESTAMP,
  resolved_by VARCHAR,
  pre_entropy FLOAT,
  post_entropy FLOAT,
  delta_entropy FLOAT,
  cls_angle FLOAT,
  semantic_polarity FLOAT,
  mutation_frequency FLOAT,
  weight FLOAT,
  last_accessed TIMESTAMP,
  scar_vector TEXT,
  vault_id VARCHAR
)
```
